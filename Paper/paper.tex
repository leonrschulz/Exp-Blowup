\section{Background}
\label{sec_background}

Formulas of propositional logic are constructed from variables
(or \emph{atoms}) with the logical connectives negation $\neg$,
disjunction $\lor$, conjunction $\land$ and implication $\to$.

A \emph{literal} is a variable $x$ oder a negated variable
$\neg x$. The literals $x$ and $\neg x$ are \emph{complements}
of each other.
%The complement of a lieral $a$ is denoted by $\bar{a}$.

A formula $F$ is in \emph{negation normal form} (NNF), if $F$ contains
no implications, and negations in $F$ only occur immediately in front
of variables, i.e., $F$ is built from literals by conjunction and
disjunction only.

For a formula $F$ in NNF we define the dual formula $\bar F$ in NNF
by replacing every literal by its complement, and interchanging the
symbols $\land$ and $\lor$.
For every formula $F$ in NNF, $\bar{F}$ is equivalent to $\neg F$.

\begin{definition}
  We define the following classes of formulas:
\begin{itemize}
\item A \emph{term} is a conjunction $a_1 \lor \ldots \lor a_k$
  of literals.
\item A \emph{clause} is a disjunction $a_1 \lor \ldots \lor a_k$ von Literalen.
\item A formula in \emph{disjunctive normal form} (DNF) is a
  disjunction $T_1 \lor \ldots \lor T_m$ of terms.
\item A formula in \emph{conjunctive normal form} (CNF) is a
  conjunction $C_1 \land \ldots \land C_m$ of clauses.
\end{itemize}
\end{definition}

Note that every formula in DNF or CNF is also in NNF.

Every formula can be transformed into an equivalent formula in DNF,
and in DNF, but the cost of this transformation is an exponential
increase in size in the worst case.

\begin{theorem} \label{dnfblowup}
  For all sufficiently large $n\in\Nat$, there are formulas $F_n$ in
  CNF with $|F_n| = O(n)$, such that every equivalent formula $G_n$ in
  DNF is of size $|G_n| \geq n2^n$.
\end{theorem}
\begin{proof}
The formulas $F_n$ are defined as $F_n = \bigwedge_{i=1}^n ( x_{i,0} \lor x_{i,1} ) \land (\neg
x_{i,0} \lor \neg x_{i,1})$.

If $G_n = \bigvee_{j\leq m} T_j$ is a formula in CNF equivalent to $F_n$, then the following two claims hold, which together prove the statement.

Claim 1. Each of the terms $T_j$ contains exactly one of the literals $x_{i,0}$ and $x_{i,1}$, for every $i\leq n$. In particular, every term $T_j$ is of size at least $n$.

Claim 2. For every vector
$\epsilon = (\epsilon_1 , \ldots ,\epsilon_n) \in \{ 0,1 \}^n$,
there is one of the terms $T_j$ in  $G_n$, which contains the literals
$x_{1,\epsilon_1}, \ldots , x_{n,\epsilon_n}$.
In particular, $G_n$ has at least $2^n$ many terms.
\end{proof}

\begin{corollary} \label{cnfblowup}
  For all sufficiently large $n\in\Nat$, there are formulas $F'_n$ in
  DNF with $|F'_n| = O(n)$, such that every equivalent formula $G_n$
  in CNF is of size $|G_n| \geq n2^n$.
\end{corollary}

\begin{proof}
  Let $F'_n := {\bar F}_n$ for the formulas $F_n$ of Theorem~\ref{dnfblowup}.
  If there were formulas  $G_n \equiv F'_n$ in CNF with $G_n < n2^n$,
  then ${\bar G}_n \equiv F_n$ would be a formula in DNF with
  $|{\bar G}_n| = |G_n| < n2^n$, contradicting Theorem~\ref{dnfblowup}.
\end{proof}

\section{Preliminaries}
\label{sec_preliminaries}

%TODO: introduce BigOr' and BigAnd' with the notation

A \emph{formula} is inductively defined as either an atom, falsehood ($\bot$), the negation of a formula, or the disjunction, conjunction or implication of two formulas.
A \emph{literal} is a special case of a formula consisting of either a positive or negative atom, or a positive or negative $\bot$.
%TODO: introduce notation x_{i,b} for i :: nat and b :: bool
%TODO: rephrase to use the notation instead of the type var
All atoms are of the type \textsf{var}, representing a combination of a Boolean value and a natural number.
A formula is \emph{conjunctive} if it is either a literal or its first subformula is a literal and its second a conjunctive formula.
Analogously a formula is \emph{disjunctive} if it is either a literal or its first subformula is a literal and its second a disjunctive formula.

A \emph{valuation} is a function from atoms to Boolean values.
The \emph{evaluation} of a formula w.r.t.\ a valuation is the Boolean value we get after replacing all atoms with the specific truth values assigned by the valuation.
We say that a valuation $\alpha$ \emph{satisfies} a formula $\phi$, written $\alpha \models \phi$, if the formula evaluates to true w.r.t.\ the valuation.
A formula $\phi$ is
\emph{satisfiable} if there exists a valuation that satisfies it (i.e., $\existsOne{\alpha}{\alpha \models \phi}$) and
\emph{tautological} if it is satisfied by every valuation (i.e., $\forallOne{\alpha}{\alpha \models \phi}$).
Two formulas $\phi$ and $\psi$ are \emph{equivalent} if they have the same truth value for every valuation (i.e., $\forallOne{\alpha}{\alpha \models \phi \iff \alpha \models \psi}$).

\begin{definitionx}
  \label{def_set}
  The function $\tyAnnot{\cst{set}}{\tyList{\yA} \Rightarrow \tySet{\yA}}$ converts a list into a set.
\end{definitionx}

\begin{definitionx}
  \label{def_sizef}
  The function $\tyAnnot{\cstSizef}{\tyFormula{\yA} \Rightarrow \tyNat}$ counts the number of atoms, $\bot$, disjunctions, conjunctions, and implications in a formula;
  the negations are ignored.
\end{definitionx}

%TODO: define the F_n function and consider lemmas not_sat_Fn_both_true and not_sat_conj_pos_false
\begin{definitionx}
  \label{def_Fn}
  The function $\tyAnnot{\cst{Fn}}{\tyNat \Rightarrow \tyFormula{\yA}}$ creates a formula containing
  the conjunction of the exclusive disjunctions of each variable pair $x_{i,false}$ and $x_{i,true}$ with $i \in {1,\ldots,n}}$:
  \begin{align*}
    \isaAp{\cst{Fn}}{0} =\; & \neg\bot \\
    \isaAp{\cst{Fn}}{(n+1)} =\; & ((x_{i,false} \lor x_{i,true}) \land (\neg x_{i,false} \lor \neg x_{i,true})) \land (\isaAp{\cst{Fn}}{n})
  \end{align*}
  In the rest of this paper, we write $F_n$ for $\isaAp{\cst{Fn}}{n}$.
\end{definitionx}

\begin{definitionx}
  \label{def_dualize}
  %TODO: tune spacing around type parameters here.
  %TODO: be consistent with abbreviations: Either always abbreviated or never
  The function $\tyAnnot{\cst{dualize}}{\tyFormula{\yA} \Rightarrow \tyFormula{\yA}}$ converts a formula in negation normal form to its dualized form by replacing each literal with its complementary:
  an atom $A$ becomes $\neg A$,
  $\bot$ becomes $\neg\bot$,
  a negation $\neg \phi$ becomes $\phi$,
  $\phi \lor \psi$ becomes $\isaAp{\cst{dualize}}{\phi} \land \isaAp{\cst{dualize}}{\psi}$, and
  $\phi \land \psi$ becomes $\isaAp{\cst{dualize}}{\phi} \lor \isaAp{\cst{dualize}}{\psi}$.
\end{definitionx}

\begin{lemma}
  \label{thm_equiv_dualize}
  Let $\phi$ be a formula.
  If $\phi$ is in negation normal form, then $\isaAp{\cst{dualize}}{\phi}$ is equivalent to $\neg\phi$.
\end{lemma}


\section{CNF to DNF}
\label{sec_CNF_to_DNF}

\begin{lemma}
  \label{aux_lemma_prop4_step1}
  Let $\phi$ be a conjunctive formula.
  Let $\alpha$ and $\beta$ be valuations.
  If both valuations map every variable occuring in $\phi$ to the same Boolean value and
  if $\alpha$ satisfies $\phi$,
  then $\beta$ also satisfies $\phi$.
\end{lemma}

\begin{lemma}
  \label{aux_lemma_prop4_step3}
  Let $\phi$ be a conjunctive formula.
  Let $\alpha$ be a valuation.
  If
  \begin{itemize}
    \item there exists a variable that occurs positively in $\phi$ and is mapped to false by $\alpha$ or
    \item there exists a variable that occurs negatively in $\phi$ and is mapped to true by $\alpha$,
  \end{itemize}
  then $\alpha$ does not satisfy $\phi$.
\end{lemma}


\begin{theorem}[TODO: Find Nice name (e.g., Weak ...)]
  \label{thm_cnf_to_dnf_blowup}
  Let $n > 0$ be a natural number.
  Let $Ts$ be a list of formulas.
  Let $G = \bigvee_{i = 0}^{\listLength{Ts}} \listNth{Ts}{i}$.
  If all formulas in $Ts$ are conjunctive and satisfiable, and
  if $F_n$ is equivalent to $G$,
  then $\isaAp{\cstSizef}{G} \ge n \cdot 2^n$.
\end{theorem}
\begin{proof}
  We proceed with five steps:
  \textbf{(1)}~First, we show that every conjunctive formula $T \in \isaAp{\cst{set}}{Ts}$ contains for all $i \in \{1,\ldots,n\}$ either the variable $x_{i,0}$ or $x_{i,1}$ positively.
  We prove this step by negation, where we have to derive false for two cases: either the variables $x_{i,0}$ and $x_{i,1}$ are both positively contained in $T$ or they are both not positively contained.
  In both cases, we proceed by defining a specific valuation that satisfies $G$ but not $F_n$.
  For instance, in the second case, we modify a valuation that satisfies $T$ by changing the assignment of the variables $x_{i,0}$ and $x_{i,1}$ to false.
  We then show using \cref{aux_lemma_prop4_step1,aux_lemma_prop4_step3} that this modified valuation satisfies $T$ (and thus $G$) but does not satisfy $F_n$.
  This means that $F_n$ is not equivalent to $G$, which contradicts \cref{thm_cnf_to_dnf_blowup}'s assumption.
  \textbf{(2)}~From step 1, we derive that every $T \in \isaAp{\cst{set}}{Ts}$ has $\isaAp{\cstSizef}{T} \ge n$.
  \textbf{(3)}~Furthermore, we prove that for every Boolean list $\epsilon = [\epsilon_1,\ldots,\epsilon_n]$ there exists a $T \in \isaAp{\cst{set}}{Ts}$ that contains the corresponding variables $x_{1,\epsilon_1},\ldots,x_{n,\epsilon_n}$ positively.
  If this is not the case for a specific list $\epsilon$, we show using step 1 that every $T$ must contain one of the variables $x_{1,\neg\epsilon_1},\ldots,x_{n,\neg\epsilon_n}$ positively.
  We then define a concrete valuation that maps all variables $x_{i,\epsilon_i}$ to true and $x_{i,\neg\epsilon_i}$ to false for $i \in \{1,\ldots,n\}$.
  We show using \cref{aux_lemma_prop4_step3} that this valuation satisfies $F_n$ but not $G$.
  This means that $F_n$ is not equivalent to $G$, which contradicts \cref{thm_cnf_to_dnf_blowup}'s assumption.
  \textbf{(4)}~From step 1, step 3, and the fact that there are $2^n$ Boolean lists of length $n$, we derive that $\listLength{Ts} \ge 2^n$.
  \textbf{(5)}~Finally, we show that $\isaAp{\cstSizef}{G} \ge n \cdot 2^n$ by combining the results from steps 2 and 4.
\end{proof}

\begin{corollary}[The actuall thing]
  \label{thm_cnf_to_dnf_blowup_math}
  Let $n$ be a natural number.
  There exists a formula $\phi$ in CNF with $\isaAp{\cstSizef}{\phi} = 8 \cdot n + 1$ such that every equivalent formula $\psi$ in DNF has $\isaAp{\cstSizef}{\psi} \ge n \cdot 2^n$.
\end{corollary}
\begin{proof}
  We start by a case analysis on $n$.
  The interesting case is when $n > 0$.
  We define $\phi = F_n$ (see \cref{def_Fn});
  it is easy to show that $F_n$ is in conjunctive normal form, and that $\isaAp{\cstSizef}{F_n} = 8 \cdot n + 1$.
  To prove the last property, we assume that there is an equivalent formula $\psi$ in disjunctive normal form.
  We first obtain a list $Ts$ of all terms in $\psi$ such that
  all formulas in $Ts$ are conjunctive and satisfiable,
  %TODO: think whether we could explain the trick with filter and undnf. Maybe extract to its own lemma?
  $\isaAp{\cstSizef}{\psi} \ge \isaAp{\cstSizef}{(\bigvee_{i = 0}^{\listLength{Ts}} \listNth{Ts}{i})}$,
  and $\bigvee_{i = 0}^{\listLength{Ts}} \listNth{Ts}{i}$ is equivalent to $\psi$.
  From this last property, we show that $\bigvee_{i = 0}^{\listLength{Ts}} \listNth{Ts}{i}$ is equivalent to $F_n$.
  This allows us to prove $\isaAp{\cstSizef}{(\bigvee_{i = 0}^{\listLength{Ts}} \listNth{Ts}{i})} \ge n \cdot 2^n$ using \cref{thm_cnf_to_dnf_blowup}.
  Finally, we can prove that $\isaAp{\cstSizef}{\psi} \ge n \cdot 2^n$ by transitivity.
\end{proof}

\section{DNF to CNF}
\label{sec_DNF_to_CNF}

\begin{lemma}
	\label{aux_lemma_cor5_step1}
	Let $\phi$ be a formula.
	If $\phi$ is in negation normal form,
	then $\isaAp{\cstSizef}{(\isaAp{\cst{dualize}}{\phi})} = \isaAp{\cstSizef}{\phi}$ holds.
\end{lemma}

\Cref{aux_lemma_cor5_step1} only holds because \cref{def_sizef} ignores negations.

\begin{lemma}
  \label{thm_dualized_disj_not_taut_impl_sat}
	Let $\phi$ be a formula.
	If $\phi$ is disjunctive and not tautological,
	then $\isaAp{\cst{dualize}}{\phi}$ is satisfiable.
\end{lemma}

\begin{lemma}
  \label{thm_dualized_conj_of_disjs_is_disj_of_conjs}
  Let $Cs$ be a list of formulas.
  If all formulas in $Cs$ are disjunctive and not tautological,
  then there exists a list $Ts$ such that all formulas in $Ts$ are conjunctive and satisfiable, and $\isaAp{\cst{dualize}}{(\bigwedge_{i = 0}^{\listLength{Cs}} \listNth{Cs}{i})} = \bigvee_{i = 0}^{\listLength{Ts}} \listNth{Ts}{i}$.
\end{lemma}
\begin{proof}
  We prove this by structural induction over the list $Cs$ using \cref{thm_dualized_disj_not_taut_impl_sat}.
  The idea is that each element of $Cs$ gets dualized.
\end{proof}

\begin{theorem}[TODO: Find Nice Name]
  Let $n > 0$ be a natural number.
  Let $Cs$ be a list of formulas.
  Let $G = \bigwedge_{i = 0}^{\listLength{Cs}} \listNth{Cs}{i}$.
  If all formulas in $Cs$ are disjunctive and not tautological, and $\isaAp{\cst{dualize}}{F_n}$ is equivalent to $G$,
  then $\isaAp{\cstSizef}{G} \ge n \cdot 2^n$.
\end{theorem}
\begin{proof}
  We prove this by contradiction using \cref{thm_cnf_to_dnf_blowup}.
  We assume $\isaAp{\cstSizef}{G} < n \cdot 2^n$ and from that derive false in five steps.
  \textbf{(1)}~First, we know from \cref{aux_lemma_cor5_step1} that $\isaAp{\cstSizef}{(\isaAp{\cst{dualize}}{G})} = \isaAp{\cstSizef}{G}$, thus $\isaAp{\cstSizef}{(\isaAp{\cst{dualize}}{G})} < n \cdot 2^n$.
  \textbf{(2)}~Moreover, we show that $\isaAp{\cst{dualize}}{G}$ is equivalent to $F_n$.
  \textbf{(3)}~Also, we prove using \cref{thm_dualized_conj_of_disjs_is_disj_of_conjs} that there exists a list $Ts$, where every formula contained in $Ts$ is both conjunctive and satisfiable, such that $\isaAp{\cst{dualize}}{G}$ is syntactically equal to $\bigvee_{i = 0}^{\listLength{Ts}} \listNth{Ts}{i}$.
  \textbf{(4)}~From steps 2 and 3, it follows that \cref{thm_cnf_to_dnf_blowup} holds for $F_n$ and $\isaAp{\cst{dualize}}{G}$.
  According to \cref{thm_cnf_to_dnf_blowup}, it therefore must be the case that $\isaAp{\cstSizef}{(\isaAp{\cst{dualize}}{G})} \ge n \cdot 2^n$.
  \textbf{(5)}~The result of step 1 contradicts the result of step 4, completing the proof.
\end{proof}

\begin{corollary}[The actuall thing]
  \label{thm_dnf_to_cnf_blowup_math}
  Let $n$ be a natural number.
  There exists a formula $F_n$ in DNF with $\isaAp{\cstSizef}{F_n} = 8 \cdot n + 1$ such that every equivalent formula $G_n$ in CNF has $\isaAp{\cstSizef}{G_n} \ge n \cdot 2^n$.
\end{corollary}
\begin{proof}
  The proof is analogous to the proof of \cref{thm_cnf_to_dnf_blowup_math}.
\end{proof}
