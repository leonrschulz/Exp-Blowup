\label{sec_CNF_to_DNF}

\begin{theorem}
  Let $n > 0$ be a natural number.
  If$F = Fn n$,
  $G = \bigvee_{t \in \textss{set} \, Ts} T$,
  $\forall T \in \textss{set} \, Ts. \textss{is\_conj} \, T \land (\exists Val. Val \models T)$,
  $\textss{equiv} \, F \, G$,
  then $\textss{sizef} \, G \ge n \cdot 2^n$.
\end{theorem}
